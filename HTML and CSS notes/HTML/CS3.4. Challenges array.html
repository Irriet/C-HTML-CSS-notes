Removing null elements from an array

string[] numbers = { "1", "2", "3", "4", "5"};
Array.Resize(ref numbers, 7);

Console.WriteLine("Resized array:");

foreach (string num in numbers)
{
 Console.WriteLine($"-- {num}");
}
int counter = 0;
for (int i = 0; i < numbers.Length; i++)
 if (numbers[i] != null)
 {
  counter++;
 }

string[] noNullArray = new string[counter];
for (int i = 0; i < noNullArray.Length; i++)
{
 if (numbers[i] != null)
  noNullArray[i] = numbers[i];
 else
  continue;
}

Console.WriteLine("New array:");

foreach(string element in noNullArray)
{
 Console.WriteLine($"-- {element}");
}
