Get absolute value of a number

static void Main(string[] args)
{
 Console.WriteLine($"The absolute value is {GetAbsoluteValue(-86)}.");
}

public static int GetAbsoluteValue(int number)
{
 if (number >= 0)
 return number;
 else
 return -number;
}

The absolute value is 86.
