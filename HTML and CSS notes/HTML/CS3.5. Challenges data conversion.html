Data Conversion Challenges

.TryParse and segregating array elements

string[] values = { "12,3", "45", "ABC", "11", "DEF" }; decimal total = 0m;
string msg = "";

foreach (var value in values)
{
 decimal result;
 if (decimal.TryParse(value, out result))
 {
  total += result;
 }
 else
 {
  msg += value;
 }
}
Console.WriteLine($"The number: {total}. \nMsg: {msg}.");
