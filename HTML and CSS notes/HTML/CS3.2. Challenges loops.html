Sum of array elements

decimal[] prices = { 12.50m, 19.84m, 51.63m, 98.98m, 32.45m, 57.00m };
decimal sum = 0;
int counter = 0;

foreach (decimal num in prices)
{
 sum += num;
 counter++;
 Console.WriteLine($"Item nr {counter}: {num}$, sum: {sum}$");
}
Console.WriteLine($"Total is: {sum}$.");
